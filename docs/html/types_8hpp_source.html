100 struct MemberTypeToSqlType {

102 static constexpr sql_type_t findType() {
103 using type = typename remove_optional<T>::type;
104 if constexpr (IsArithmetic<type>()) {
105 if (std::is_floating_point_v<type>) {
106 return sql_type_t::REAL;
107 } else {
108 return sql_type_t::INTEGER;
109 }
110 } else if constexpr (IsString<type>()) {
111 return sql_type_t::TEXT;
112 } else if constexpr (IsContinuousContainer<type>()) {
113 return sql_type_t::BLOB;
114 } else {
115 static_assert(std::is_same_v<T, std::false_type>,
116 "Member type is not convertable to an sql type. "
117 "Please use only arithmetic types or containers with continuous memory, "
118 "or std::optional containing either type");
119 }
120 }
121 public:
